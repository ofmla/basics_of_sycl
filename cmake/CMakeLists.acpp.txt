
cmake_minimum_required (VERSION 3.5)
set(CMAKE_CXX_COMPILER "acpp")

project(SyclAcpp CXX)

set(MY_CXX_FLAGS "-std=c++20 -O3 -Wall -Wextra -fno-fast-math -ffp-contract=fast -march=native -mtune=native -g")
set(MY_SYCL_FLAGS "")

set(COMPILE_FLAGS "${MY_CXX_FLAGS} ${MY_SYCL_FLAGS}")
set(LINK_FLAGS    "${MY_CXX_FLAGS} ${MY_SYCL_FLAGS}")

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})

add_subdirectory (src)
